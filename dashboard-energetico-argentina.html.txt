        // Navigation
        function showSection(sectionId) {
            // Hide all sections
            document.querySelectorAll('.section').forEach(section => {
                section.classList.remove('active');
            });
            
            // Remove active class from all nav buttons
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            
            // Show selected section
            document.getElementById(sectionId).classList.add('active');
            
            // Add active class to clicked button
            document.querySelector(`[data-section="${sectionId}"]`).classList.add('active');
            
            // Initialize charts when sections are shown
            if (sectionId === 'graficos') {
                setTimeout(initCharts, 100);
            } else if (sectionId === 'produccion') {
                setTimeout(initProductionChart, 100);
            }
        }

        // Initialize charts

        // Companies
        function renderCompanies() {
            const grid = document.getElementById('companiesGrid');
            const noResults = document.getElementById('noResults');
            
            if (filteredCompanies.length === 0) {
                grid.innerHTML = '';
                noResults.classList.remove('hidden');
                return;
            }
            
            noResults.classList.add('hidden');
            
            grid.innerHTML = filteredCompanies.map(company => `
                <div class="company-card" onclick="openModal('${company.symbol}')">
                    <div class="company-header">
                        <div>
                            <h3 class="company-name">${company.name}</h3>
                            <p class="company-symbol">${company.symbol}</p>
                            <div class="stars">
                                ${renderStars(company.rating)}
                                <span style="color: #9ca3af; font-size: 0.75rem; margin-left: 0.25rem;">(${company.rating})</span>
                            </div>
                        </div>
                        <span class="company-badge">${company.sector}</span>
                    </div>
                    <div class="company-price">
                        <span class="price">$${company.price}</span>
                        <div class="change ${company.change >= 0 ? 'positive' : 'negative'}">
                            <i class="fas fa-arrow-${company.change >= 0 ? 'up' : 'down'}"></i>
                            <span>${company.change >= 0 ? '+' : ''}${company.change}%</span>
                        </div>
                    </div>
                    <div style="display: flex; justify-content: space-between; font-size: 0.875rem; color: #9ca3af; margin-top: 0.5rem;">
                        <span>Cap. Mercado: ${company.marketCap}</span>
                        <span>Volumen: ${company.volume}</span>
                    </div>
                </div>
            `).join('');
        }

        function renderStars(rating) {
            let stars = '';
            for (let i = 1; i <= 5; i++) {
                stars += `<i class="fas fa-star star ${i <= Math.floor(rating) ? 'filled' : ''}"></i>`;
            }
            return stars;
        }

        function filterCompanies() {
            const searchTerm = document.getElementById('searchInput').value.toLowerCase();
            filteredCompanies = companies.filter(company =>
                company.name.toLowerCase().includes(searchTerm) ||
                company.symbol.toLowerCase().includes(searchTerm)
            );
            renderCompanies();
        }
